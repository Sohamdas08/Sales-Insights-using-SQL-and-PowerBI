TOP 5 Products by Revenue


SELECT 
    p.product_id,
    p.product_name,
    p.category,
    SUM(s.selling_price * s.quantity) AS total_revenue
FROM sales s
JOIN products p ON s.product_id = p.product_id
GROUP BY p.product_id, p.product_name, p.category
ORDER BY total_revenue DESC
LIMIT 5;
